pipeline {
    agent any

    environment {
        DEPLOY_DIR  = '/home/ubuntu/API/'		// Папка на цели
        SERVICE     = 'DeviceManager.service'	// Имя службы
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build & Publish') {
            steps {
                // -r linux-x64: целевая платформа
                // --self-contained false: если на цели установлен .NET Runtime 3.1
                sh 'dotnet publish -c Release -r linux-x64 --self-contained false -o ./linux-x64'
            }
        }

		stage('Clean...') {
            steps {
				sh 'rm -rf $DEPLOY_DIR/* && mkdir -p $DEPLOY_DIR'
			}
		}
		stage('Move build') {
            steps {
				sh 'mv -f ./linux-x64/* $DEPLOY_DIR'
			}
		}
		stage('Restart Service') {
			steps {
				sh 'sudo chmod 777 $DEPLOY_DIR -R'
				sh 'sudo systemctl restart $SERVICE'
			}
		}
		stage('Get status Service') {
			steps {
				sh '''
					sleep 1
					
					for i in 1 2 3 4 5; do
						if sudo systemctl is-active --quiet $SERVICE; then
							echo "✅ Service is active (running)"
							exit 0
						fi
							echo "⏳ Waiting... ($i)"
						sleep 2
					done
					
					echo "❌ Service failed to start"
						sudo systemctl status $SERVICE --no-pager
						sudo journalctl -u $SERVICE --no-pager -n 20
					exit 1
				'''
			}
		}
    }
}
